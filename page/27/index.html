def recursive_binary_search(arr, low, high, key):
        if low <= high:
                mid = (low + high) // 2
            if arr[mid] == key:
                    return mid
            elif key < arr[mid]:
                    return recursive_binary_search(arr, low, mid-1, key)
            else:
                    return recursive_binary_search(arr, mid+1, high, key)

        return -1
